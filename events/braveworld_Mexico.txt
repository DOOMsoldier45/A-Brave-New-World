add_namespace = mexico


#Mexican civil war
country_event={
	
	id = mexico.1
	title = mexico.1.t
	desc  = mexico.1.d
	
	fire_only_once = yes
	
	trigger = {
		date > 1910.11.10
	}
	
	#Step down
	option = {
		name = mexico.1.a
		
		retire_country_leader = yes
		create_country_leader = {
			name = "Francisco I. Madero"
			expire = "1940.1.1"
			#picture = "Portrait_China_Puyi.dds"
			ideology = liberalism
			traits = {
				inexperienced_politician
			}
		}
		
		remove_ideas = old_man_dictator
		add_ideas = improvement_over_diaz
		
	}
	
	#Revolution
	option = {
		name = mexico.1.b
		hidden_effect = {
			start_civil_war = {
				ideology = democratic
				capital = 482
				states = { 482 480 476 }
			}
			
			random_other_country = {
				limit = {
					original_tag = MEX
					has_government = democratic
				}
				
				create_country_leader = {
					name = "Francisco I. Madero"
					expire = "1940.1.1"
					#picture = "Portrait_China_Puyi.dds"
					ideology = liberalism
					traits = {
						inexperienced_politician
						rule_in_exile
					}
				}
				
				set_equipment_fraction = 0.8
				set_national_unity = 0.9
				load_oob = "Mex_1911_Rev"
			}
		}
		custom_effect_tooltip = mex.1.tt
		set_country_flag = mexican_revolution_1
	}
}
