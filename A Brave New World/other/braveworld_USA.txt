	id = usa.31

	desc = usa.31.d
	picture = GFX_report_event_usa_macarthur_speech

	is_triggered_only = yes
	
	#Intervene ultimatum war
	option = { 
		name = usa.31.a
	
		#Need to use FROM FROM scope since the one sending the event is the nation under attack
		ai_chance = {
			factor = 20			
			modifier = {
				factor = 0.2
				FROM = {
					FROM = {
						strength_ratio = {
							tag = USA
							ratio > 0.6
						}
					}
				}
			}		
		}

		custom_effect_tooltip = usa.31.a_tt
		FROM = { FROM = { country_event = { id = usa.31 days = 2 } } }
	}

	#Don't Intervene
	option = {
		name = usa.31.b

		ai_chance = {
			factor = 5
			modifier = {
				factor = 0
				FROM = {
					FROM = {
						strength_ratio = {
							tag = USA
							ratio < 0.3
						}
					}
				}
			}
		}
	}
}

				}
			}

		}

			LIMIT={
				tag = PAN

		}
	}
	
}
